public function checkData(){
	$AK = $this->getUser();
	$AK = new pPArbeitskraft($AK->getID());

	$return = array();
	foreach($this->requiredFields AS $field){
		if($field == "pPArbeitskraftKinder")
			continue;

		if(trim($AK->A($field)) == "" OR trim($AK->A($field)) == "0")
			$return[] = $field;
	}

	if($AK->A("pPArbeitskraftKinder") == "-1")
		$return[] = "pPArbeitskraftKinder";

	return $return;
}
